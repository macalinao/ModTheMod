017package org.thedevteam.modthemod.part;
018
019import java.util.HashMap;
020import java.util.Map;
021
022/**
023 * Holds properties of a part.
024 */
025public class PartProperties {
026
027    /**
028     * The properties of the part.
029     */
030    private Map<String, String> properties = new HashMap<String, String>();
031
032    /**
033     * Constructor.
034     */
035    public PartProperties() {
036    }
037
038    /**
039     * Adds a property to this {@link PartProperties} object.
040     *
041     * @param name The name of the property.
042     * @param value The value of the property.
043     */
044    public void addProperty(String name, String value) {
045        properties.put(name, value);
046    }
047
048}
