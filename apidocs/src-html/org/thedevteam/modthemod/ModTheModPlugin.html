017package org.thedevteam.modthemod;
018
019import org.thedevteam.modthemod.mod.ModManager;
020import org.spout.api.plugin.CommonPlugin;
021import static org.thedevteam.modthemod.MLogger.info;
022
023/**
024 * ModTheMod main plugin file.
025 */
026public class ModTheModPlugin extends CommonPlugin {
027
028    private ModManager modManager;
029
030    @Override
031    public void onEnable() {
032        //Setup logging
033        MLogger.setLogger(getLogger());
034
035        //Start enabling
036        info("=============== MOD THE MOD ===============");
037        info("=== Setting up the mod manager...");
038        modManager = new ModManager(this);
039        info("=========== MOD THE MOD ENABLED! ==========");
040        //Stop enabling
041    }
042
043    @Override
044    public void onDisable() {
045        //Start disabling
046        info("=============== MOD THE MOD ===============");
047        info("========== MOD THE MOD DISABLED! ==========");
048        //Stop disabling
049
050        //Unload logging.
051        MLogger.setLogger(null);
052    }
053
054    public ModManager getModManager() {
055        return modManager;
056    }
057
058}
