017package org.thedevteam.modthemod.mod.js;
018
019import com.google.common.base.Charsets;
020import org.mozilla.javascript.Callable;
021import org.mozilla.javascript.Context;
022import org.mozilla.javascript.Scriptable;
023import org.mozilla.javascript.ScriptableObject;
024import org.thedevteam.modthemod.mod.Mod;
025import org.thedevteam.modthemod.mod.ModLoader;
026import org.thedevteam.modthemod.mod.ModLanguage;
027
028/**
029 * Represents a Javascript mod loader.
030 */
031public class JSModLoader implements ModLoader {
032
033    @Override
034    public Mod loadMod(byte[] data) {
035        String js = new String(data, Charsets.UTF_8);
036
037        try {
038            Context cx = Context.enter();
039            Scriptable scope = cx.initStandardObjects();
040            
041            //Define mod
042            scope.put("defineMod", scope, new Callable() {
043
044                @Override
045                public Object call(Context cx, Scriptable scope, Scriptable thisObj, Object[] args) {
046                    if (args.length < 1) {
047                        return false;
048                    }
049                    
050                    if (!(args[0] instanceof Scriptable)) {
051                        return false;
052                    }
053                    
054                    Scriptable modData = (Scriptable) args[0];
055                    
056                    //Name
057                    Object nameObj = modData.get("name", modData);
058                    if (nameObj == null) {
059                        return false;
060                    }
061                    String name = nameObj.toString();
062                    
063                    //Description
064                    Object descObj = modData.get("description", modData);
065                    if (!(descObj instanceof Scriptable)) {
066                        return false;
067                    }
068                    Scriptable desc = (Scriptable) descObj;
069                    
070                    
071                    
072                    //Parts
073                    
074                    return null;
075                }
076
077            });
078
079            //Evaluate
080            cx.evaluateString(scope, js, "<mod>", 1, null);
081        } finally {
082            Context.exit();
083        }
084
085        return null;
086    }
087
088    @Override
089    public final ModLanguage getType() {
090        return ModLanguage.JAVASCRIPT;
091    }
092
093}
