001    package com.modthemod.engine.property;
002    
003    import java.util.HashMap;
004    import java.util.Map;
005    import java.util.logging.Level;
006    import java.util.regex.Pattern;
007    
008    import com.modthemod.api.base.Base;
009    import com.modthemod.api.mod.Mod;
010    import com.modthemod.api.property.Matcher;
011    import com.modthemod.api.property.Primitive;
012    import com.modthemod.api.property.Type;
013    import com.modthemod.api.property.TypeManager;
014    import com.modthemod.engine.MGame;
015    
016    public class MTypeManager implements TypeManager {
017            private static final Pattern basePattern = Pattern
018                            .compile("\\(([a-z,]+)\\)");
019    
020            private final MGame game;
021    
022            private Map<String, Primitive<?>> primitives = new HashMap<String, Primitive<?>>();
023    
024            private Map<Mod, Map<String, Base>> bases = new HashMap<Mod, Map<String, Base>>();
025    
026            private Map<String, Base> baseNames = new HashMap<String, Base>();
027    
028            public MTypeManager(MGame game) {
029                    this.game = game;
030            }
031    
032            @Override
033            public Matcher getMatcher(String name) {
034                    // TODO Auto-generated method stub
035                    return null;
036            }
037    
038            @Override
039            public Type<?> getType(String name) {
040                    String baseName = basePattern.matcher(name).group();
041                    if (!baseName.isEmpty() && baseName != null && name.matches("[a-z]+")) {
042                            return baseNames.get(baseName);
043                    }
044                    return primitives.get(name);
045            }
046    
047            @Override
048            public void registerPrimitive(Primitive<?> primitive) {
049                    primitives.put(primitive.getName(), primitive);
050            }
051    
052            @Override
053            public void registerBase(Base base) {
054                    Map<String, Base> map = getBaseMap(base.getMod());
055                    Base exists = map.get(base.getName());
056                    if (exists != null) {
057                            game.getLogger().log(
058                                            Level.WARNING,
059                                            "Base '" + exists.getName()
060                                                            + "' already exists! Be careful!");
061                    }
062                    map.put(base.getName(), base);
063    
064                    // Now the name
065                    baseNames.put(base.getMod().getName() + ":" + base.getName(), base);
066            }
067    
068            /**
069             * Gets the map of bases of the given mod.
070             * 
071             * @param mod
072             * @return
073             */
074            private Map<String, Base> getBaseMap(Mod mod) {
075                    Map<String, Base> map = bases.get(mod);
076                    if (map == null) {
077                            map = new HashMap<String, Base>();
078                            bases.put(mod, map);
079                    }
080                    return map;
081            }
082    }
