001    package com.modthemod.modthemod.event;
002    
003    import java.util.HashMap;
004    import java.util.Map;
005    import java.util.logging.Level;
006    
007    import com.modthemod.api.event.EventManager;
008    import com.modthemod.api.event.EventType;
009    import com.modthemod.api.mod.Mod;
010    import com.modthemod.modthemod.MGame;
011    
012    public class MEventManager implements EventManager {
013            private final MGame game;
014    
015            private final Map<Mod, Map<String, EventType>> eventTypes = new HashMap<Mod, Map<String, EventType>>();
016    
017            private final Map<String, EventType> eventTypeNames = new HashMap<String, EventType>();
018    
019            public MEventManager(MGame game) {
020                    this.game = game;
021            }
022    
023            @Override
024            public void registerEvent(EventType type, Mod mod) {
025                    Map<String, EventType> map = getEventMap(mod);
026                    EventType now = map.get(type.getName());
027                    if (now != null) {
028                            game.getLogger().log(
029                                            Level.WARNING,
030                                            "EventType '" + type.getName()
031                                                            + "' already exists! Be careful!!!");
032                    }
033                    map.put(type.getName(), type);
034    
035                    // Now the name
036                    eventTypeNames.put(mod.getName() + ":" + type.getName(), type);
037            }
038    
039            @Override
040            public EventType getEvent(String name) {
041                    return eventTypeNames.get(name.toLowerCase());
042            }
043    
044            /**
045             * Gets the EventType map for the given mod.
046             * 
047             * @param mod
048             * @return
049             */
050            private Map<String, EventType> getEventMap(Mod mod) {
051                    Map<String, EventType> map = eventTypes.get(mod);
052                    if (map == null) {
053                            map = new HashMap<String, EventType>();
054                            eventTypes.put(mod, map);
055                    }
056                    return map;
057            }
058    
059    }
