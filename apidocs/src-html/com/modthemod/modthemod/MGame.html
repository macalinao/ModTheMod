001    package com.modthemod.modthemod;
002    
003    import com.modthemod.api.Game;
004    import com.modthemod.api.platform.Platform;
005    import com.modthemod.modthemod.entity.MEntityManager;
006    import com.modthemod.modthemod.event.MEventManager;
007    import com.modthemod.modthemod.property.MTypeManager;
008    
009    /**
010     * Represents the game.
011     */
012    public class MGame implements Game {
013            /**
014             * The platform of the game.
015             */
016            private final Platform platform;
017    
018            private final MEntityManager entityManager;
019    
020            private final MEventManager eventManager;
021    
022            private final MTypeManager typeManager;
023    
024            public MGame(Platform platform) {
025                    this.platform = platform;
026    
027                    this.entityManager = new MEntityManager(this);
028                    this.eventManager = new MEventManager(this);
029                    this.typeManager = new MTypeManager(this);
030            }
031    
032            @Override
033            public Platform getPlatform() {
034                    return platform;
035            }
036    
037            @Override
038            public MEntityManager getEntityManager() {
039                    return entityManager;
040            }
041    
042            @Override
043            public MEventManager getEventManager() {
044                    return eventManager;
045            }
046    
047            @Override
048            public MTypeManager getTypeManager() {
049                    return typeManager;
050            }
051    
052    }
