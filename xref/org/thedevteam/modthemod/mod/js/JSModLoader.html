31  public class JSModLoader implements ModLoader {
32  
33      @Override
34      public Mod loadMod(byte[] data) {
35          String js = new String(data, Charsets.UTF_8);
36  
37          try {
38              Context cx = Context.enter();
39              Scriptable scope = cx.initStandardObjects();
40              
41              //Define mod
42              scope.put("defineMod", scope, new Callable() {
43  
44                  @Override
45                  public Object call(Context cx, Scriptable scope, Scriptable thisObj, Object[] args) {
46                      if (args.length < 1) {
47                          return false;
48                      }
49                      
50                      if (!(args[0] instanceof Scriptable)) {
51                          return false;
52                      }
53                      
54                      Scriptable modData = (Scriptable) args[0];
55                      
56                      //Name
57                      Object nameObj = modData.get("name", modData);
58                      if (nameObj == null) {
59                          return false;
60                      }
61                      String name = nameObj.toString();
62                      
63                      //Description
64                      Object descObj = modData.get("description", modData);
65                      if (!(descObj instanceof Scriptable)) {
66                          return false;
67                      }
68                      Scriptable desc = (Scriptable) descObj;
69                      
70                      
71                      
72                      //Parts
73                      
74                      return null;
75                  }
76  
77              });
78  
79              //Evaluate
80              cx.evaluateString(scope, js, "<mod>", 1, null);
81          } finally {
82              Context.exit();
83          }
84  
85          return null;
86      }
87  
88      @Override
89      public final ModLanguage getType() {
90          return ModLanguage.JAVASCRIPT;
91      }
