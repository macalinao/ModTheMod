87  	 */
88  	public Set<Mod> loadMods(File directory) {
89  		if (!directory.isDirectory()) {
90  			throw new IllegalArgumentException(
91  					"Specified mod directory is not a directory!");
92  		}
93  
94  		Set<Mod> result = new HashSet<Mod>();
95  		LinkedList<File> files = new LinkedList<File>(Arrays.asList(directory
96  				.listFiles()));
97  		boolean failed = false;
98  		boolean lastPass = false;
99  
100 		while (!failed || lastPass) {
101 			failed = true;
102 			Iterator<File> it = files.iterator();
103 
104 			while (it.hasNext()) {
105 				File file = it.next();
106 				Mod mod = null;
107 
108 				if (file.isDirectory()) {
109 					it.remove();
110 					continue; // TODO "Development mode" for mods
111 				}
112 
113 				try {
114 					mod = loadMod(file);
115 					it.remove();
116 				} catch (Exception ex) {
117 					return null; // TODO finishme
118 				}
119 			}
120 
121 		}
122 
123 		return result;
124 	}

126 	/**
127 	 * Load a mod with the given file.
128 	 * 
129 	 * @param file
130 	 * @return
131 	 */
132 	public Mod loadMod(File file) {
133 		ModLoader<?> loader = null;
134 		String fileName = file.getName();
135 
136 		for (Entry<Pattern, ModLoader<?>> entry : filters.entrySet()) {
137 			Matcher match = entry.getKey().matcher(fileName);
138 			if (match.find()) {
139 				loader = entry.getValue();
140 				break;
141 			}
142 		}
143 
144 		if (loader == null) {
145 			return null; // Don't load any mod. Ignore the file.
146 		}
147 
148 		Mod mod = null;
149 		try {
150 			mod = loader.loadMod(file);
151 		} catch (InvalidModException e) {
152 			MLogger.log(Level.SEVERE, "Invalid mod encountered!", e);
153 		}
154 
155 		return mod;
156 	}
