79  	 * Loads mods in the given directory.
80  	 * 
81  	 * @param directory
82  	 *            The directory to load mods from.
83  	 * @return
84  	 */
85  	public Set<Mod> loadMods(File directory) {
86  		if (!directory.isDirectory()) {
87  			throw new IllegalArgumentException(
88  					"Specified mod directory is not a directory!");
89  		}
90  
91  		Set<Mod> result = new HashSet<Mod>();
92  
93  		for (File file : directory.listFiles()) {
94  			ModLoader<?> loader = null;
95  
96  			for (Entry<Pattern, ModLoader<?>> entry : filters.entrySet()) {
97  				Matcher match = entry.getKey().matcher(file.getName());
98  				if (match.find()) {
99  					loader = entry.getValue();
100 					break;
101 				}
102 			}
103 
104 			if (loader == null) {
105 				continue;
106 			}
107 
108 			Mod mod = null;
109 			try {
110 				mod = loader.loadMod(file);
111 			} catch (InvalidModException e) {
112 				MLogger.log(Level.SEVERE, "Invalid mod encountered!", e);
113 			}
114 			result.add(mod);
115 		}
116 
117 		return result;
118 	}
